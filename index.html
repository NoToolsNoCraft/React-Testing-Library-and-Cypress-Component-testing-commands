Cypress vs React Testing Library: Component Testing Commands

This cheatsheet shows equivalent commands between Cypress Component Testing and React Testing Library (RTL), including the necessary imports.

üü¢ Core Commands

PurposeCypress Component TestingReact Testing Library
Render a component import { mount } from "cypress/react";
cy.mount(<Component />)		 import { render, screen } from "@testing-library/react";
render(<Component />)
Query element by text // built into Cypress
cy.contains("text")		 import { render, screen } from "@testing-library/react";
screen.getByText("text")
Query element by role import "@testing-library/cypress/add-commands";
cy.findByRole("button", { name: /submit/i })		 import { render, screen } from "@testing-library/react";
screen.getByRole("button", { name: /submit/i })
Query element by test id // built into Cypress
cy.get("[data-testid='my-element']")		 import { render, screen } from "@testing-library/react";
screen.getByTestId("my-element")
Assert element exists // built into Cypress
cy.get("selector").should("exist")		 import "@testing-library/jest-dom";
expect(screen.getByText("text")).toBeInTheDocument()
Assert element not exists // built into Cypress
cy.get("selector").should("not.exist")		 import "@testing-library/jest-dom";
expect(screen.queryByText("text")).not.toBeInTheDocument()
Interact: click // built into Cypress
cy.get("button").click()		 import { fireEvent } from "@testing-library/react";
fireEvent.click(screen.getByRole("button"))

// OR with userEvent
import userEvent from "@testing-library/user-event";
userEvent.click(screen.getByRole("button"))
Interact: type text // built into Cypress
cy.get("input").type("Hello")		 import { fireEvent } from "@testing-library/react";
fireEvent.change(screen.getByRole("textbox"), { target: { value: "Hello" }})

// OR with userEvent
import userEvent from "@testing-library/user-event";
userEvent.type(screen.getByRole("textbox"), "Hello")
Check element value // built into Cypress
cy.get("input").should("have.value", "Hello")		 import "@testing-library/jest-dom";
expect(screen.getByRole("textbox")).toHaveValue("Hello")
Wait for async updates // built into Cypress (auto-retry)
cy.contains("Loaded")		 import { render, screen } from "@testing-library/react";
await screen.findByText("Loaded")
Run code before/after test // Mocha built-in (no extra import)
beforeEach(() => {...})
afterEach(() => {...})		 // Jest built-in (no extra import)
beforeEach(() => {...})
afterEach(() => {...})
